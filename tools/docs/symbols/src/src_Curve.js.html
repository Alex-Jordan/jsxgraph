 30 /**
 31  * Curves are the common object for function graphs, parametric curves, polar curves, adn data plots.
 32  * @class Creates a new curve object. Do not use this constructor to create a curve. Use {@link JXG.Board#create} with
 33  * type {@link Curve}, or {@link Functiongraph} instead.  
 34  * @augments JXG.GeometryElement
 35  * @param {string,JXG.Board} board The board the new curve is drawn on.
 36  * @param {Array} defining terms An array with the functon terms, data points of the curve.
 37  * @param {String} id Unique identifier for the point. If null or an empty string is given,
 38  *  an unique id will be generated by Board
 39  * @param {String} name Not necessarily unique name for the point. If null or an
 40  *  empty string is given, an unique name will be generated
 41  * @param {boolean} show False if the point is invisible, True otherwise
 42  * @see JXG.Board#generateName
 43  * @see JXG.Board#addCurve
 44   */
 45 JXG.Curve = function (board, parents, id, name, withLabel, layer) {
 46     this.constructor();
 47  
 48     this.points = []; 
 49 
 50     this.type = JXG.OBJECT_TYPE_CURVE;
 51     this.elementClass = JXG.OBJECT_CLASS_CURVE;                
 52     
 53     this.init(board, id, name);
 54 
 55     /**
 56      * Set the display layer.
 57      */
 58     if (layer == null) layer = board.options.layer['curve'];
 59     this.layer = layer;
 60 
 61     /** Use the algorithm by Gillam and Hohenwarter for plotting.
 62       * If false the naive algorithm is used.
 63       * It is much slower, but the result is better.
 64       */
 65     this.doAdvancedPlot = this.board.options.curve.doAdvancedPlot;
 66     
 67     /** 
 68       * Number of points on curves after mouseUp, i.e. high quality output.
 69       * Only used if this.doAdvancedPlot==false
 70       * May be overwritten.
 71       **/
 72     this.numberPointsHigh = this.board.options.curve.numberPointsHigh;
 73     /** 
 74       * Number of points on curves after mousemove, i.e. low quality output.
 75       * Only used if this.doAdvancedPlot==false
 76       * May be overwritten.
 77       **/
 78     this.numberPointsLow = this.board.options.curve.numberPointsLow;
 79     /** 
 80       * Number of points on curves. This value changes
 81       * between numberPointsLow and numberPointsHigh.
 82       * It is set in {@link #updateCurve}.
 83       */
 84     this.numberPoints = this.numberPointsHigh; 
 85 
 86     this.visProp['strokeWidth'] = this.board.options.curve.strokeWidth;
 87     this.visProp['highlightStrokeWidth'] = this.visProp['strokeWidth'];
 88 
 89     this.visProp['visible'] = true;
 90     this.dataX = null;
 91     this.dataY = null;
 92 
 93     /**
 94      * This is just for the hasPoint() method.
 95      * @type int
 96      */
 97     //this.r = this.board.options.precision.hasPoint;
 98     
 99     /**
100      * The curveType is set in @see generateTerm and used in 
101      * {@link updateCurve}
102      * Possible values are:
103      * 'none'
104      * 'plot': Data plot
105      * 'parameter': we can not distinguish function graphs and parameter curves
106      * 'functiongraph': function graph
107      * 'polar'
108      * 'implicit' (not yet)
109      *
110      * Only parameter and plot are set directly.
111      * polar is set with setProperties only.
112      **/
113     // this.curveType = 'none';
114     this.curveType = null;
115 
116     if (parents[0]!=null) {
117         this.varname = parents[0];
118     } else {
119         this.varname = 'x';
120     }
121     this.xterm = parents[1];  // function graphs: "x"
122     this.yterm = parents[2];  // function graphs: e.g. "x^2"
123     this.generateTerm(this.varname,this.xterm,this.yterm,parents[3],parents[4]);  // Converts GEONExT syntax into JavaScript syntax
124     this.updateCurve();                        // First evaluation of the curve
125     
126     this.createLabel(withLabel);
127     this.id = this.board.addCurve(this);
128     
129     if (typeof this.xterm=='string') {
130         this.notifyParents(this.xterm);
131     }
132     if (typeof this.yterm=='string') {
133         this.notifyParents(this.yterm);
134     }
135 };

139  * Gives the default value of the left bound for the curve.
140  * May be overwritten in @see generateTerm.
141  */
142 JXG.Curve.prototype.minX = function () {
143     if (this.curveType=='polar') {
144         return 0.0;
145     } else {
146         var leftCoords = new JXG.Coords(JXG.COORDS_BY_SCREEN, [0, 0], this.board);
147         return leftCoords.usrCoords[1];
148     }

151 /**
152  * Gives the default value of the right bound for the curve.
153  * May be overwritten in @see generateTerm.
154  */
155 JXG.Curve.prototype.maxX = function () {
156     var rightCoords;
157     if (this.curveType=='polar') {
158         return 2.0*Math.PI;
159     } else {
160         rightCoords = new JXG.Coords(JXG.COORDS_BY_SCREEN, [this.board.canvasWidth, 0], this.board);
161         return rightCoords.usrCoords[1];
162     }

165 /**
166  * Checks whether (x,y) is near the curve.
167  * @param {int} x Coordinate in x direction, screen coordinates.
168  * @param {int} y Coordinate in y direction, screen coordinates.
169  * @param {y} Find closest point on the curve to (x,y)
170  * @return {bool} True if (x,y) is near the curve, False otherwise.
171  */
172 JXG.Curve.prototype.hasPoint = function (x,y) {
173     var t, dist = Infinity, 
174         c, trans, i, j, tX, tY,
175         xi, xi1, yi, yi1,
176         lbda, x0, y0, x1, y1, xy, den,
177         steps = this.numberPointsLow, 
178         d = (this.maxX()-this.minX())/steps,
179         prec = this.board.options.precision.hasPoint/(this.board.unitX*this.board.zoomX),
180         checkPoint, len,
181         suspendUpdate = true;
182 
183     prec = prec*prec;
184     checkPoint = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x,y], this.board);
185     x = checkPoint.usrCoords[1];
186     y = checkPoint.usrCoords[2];
187     if (this.curveType=='parameter' || this.curveType=='polar' || this.curveType=='functiongraph') { 
188         // Brute fore search for a point on the curve close to the mouse pointer
189         len = this.transformations.length;
190         for (i=0,t=this.minX(); i<steps; i++) {
191             tX = this.X(t,suspendUpdate);
192             tY = this.Y(t,suspendUpdate);
193             for (j=0; j<len; j++) {
194                 trans = this.transformations[j];
195                 trans.update();
196                 c = JXG.Math.matVecMult(trans.matrix,[1,tX,tY]);
197                 tX = c[1];
198                 tY = c[2];
199             }
200             dist = (x-tX)*(x-tX)+(y-tY)*(y-tY);
201             if (dist<prec) { return true; }
202             t+=d;
203         }  
204     } else if (this.curveType == 'plot') {
205         //$('debug').innerHTML +='. ';
206         len = this.numberPoints; // Rough search quality
207         for (i=0;i<len-1;i++) {
208             xi = this.X(i);
209             xi1 = this.X(i+1);
210             //if (i!=xi) {
211             //    yi = this.Y(xi);
212             //    yi1 = this.Y(xi1);
213             //} else {
214                 yi = this.Y(i);
215                 yi1 = this.Y(i+1);
216                // $('debug').innerHTML = this.Y.toString();
217             //}
218             x1 = xi1 - xi;
219             y1 = yi1-yi;
220             
221             x0 = x-xi; //this.X(i);
222             y0 = y-yi; //this.Y(i);
223             den = x1*x1+y1*y1;
224             
225             if (den>=JXG.Math.eps) {
226                 xy = x0*x1+y0*y1;
227                 lbda = xy/den;
228                 dist = x0*x0+y0*y0 - lbda*xy;
229             } else {
230                 lbda = 0.0;
231                 dist = x0*x0+y0*y0;
232             }
233             if (lbda>=0.0 && lbda<=1.0 && dist<prec) { 
234                 return true; 
235             } 
236         }
237         return false;
238     } 
239     return (dist<prec);
240 };

245 JXG.Curve.prototype.allocatePoints = function () {
246     var i, len;
247     len = this.numberPoints;
248     if (this.points.length<this.numberPoints) {
249         for (i=this.points.length; i<len; i++) {
250             this.points[i] = new JXG.Coords(JXG.COORDS_BY_USER, [0,0], this.board);
251         }
252     }

257  * of the function, {@link #updateCurve}
258  * Then, the update function of the renderer

262     if (this.needsUpdate) {

264     }

266         this.cloneToBackground(true);
267     }    
268     return this;

301 /**
302  * Computes for equidistant points on the x-axis the values
303  * of the function. @see #update
304  * If the mousemove event triggers this update, we use only few
305  * points. Otherwise, e.g. on mouseup, many points are used.
306  */
307 JXG.Curve.prototype.updateCurve = function () {
308     var len, mi, ma, x, y, i,
309         suspendUpdate = false;
310     
311     this.updateDataArray();
312     mi = this.minX();
313     ma = this.maxX();
314 
315     // Discrete data points
316     if (this.dataX!=null) { // x-coordinates are in an array
317         this.numberPoints = this.dataX.length;
318         len = this.numberPoints;
319         this.allocatePoints();  // It is possible, that the array length has increased.
320         for (i=0; i<len; i++) {
321             x = i;
322             if (this.dataY!=null) { // y-coordinates are in an array
323                 y = i;
324             } else {
325                 y = this.X(x); // discrete x data, continuous y data
326             }
327             this.points[i].setCoordinates(JXG.COORDS_BY_USER, [this.X(x,suspendUpdate),this.Y(y,suspendUpdate)], false); // The last parameter prevents rounding in usr2screen().
328             this.updateTransform(this.points[i]);
329             suspendUpdate = true;
330         }
331     } else { // continuous x data
332         if (this.doAdvancedPlot) {
333             this.updateParametricCurve(mi,ma,len);
334         } else {
335             if (this.board.updateQuality==this.board.BOARD_QUALITY_HIGH) {
336                 this.numberPoints = this.numberPointsHigh;
337             } else {
338                 this.numberPoints = this.numberPointsLow;
339             }
340             len = this.numberPoints;
341             this.allocatePoints();  // It is possible, that the array length has increased.
342             this.updateParametricCurveNaive(mi,ma,len);
343         }
344     }
345     this.getLabelAnchor();
346     return this;
347 };

349 JXG.Curve.prototype.updateParametricCurveNaive = function(mi,ma,len) {
350     var i, t,
351         suspendUpdate = false,
352         stepSize = (ma-mi)/len;
353         
354     for (i=0; i<len; i++) {
355         t = mi+i*stepSize;
356         this.points[i].setCoordinates(JXG.COORDS_BY_USER, [this.X(t,suspendUpdate),this.Y(t,suspendUpdate)], false); // The last parameter prevents rounding in usr2screen().
357         this.updateTransform(this.points[i]);
358         suspendUpdate = true;
359     }
360     return this;
361 };

363 JXG.Curve.prototype.updateParametricCurve = function(mi,ma,len) {
364     var i, t, t0,
365         suspendUpdate = false,
366         po = new JXG.Coords(JXG.COORDS_BY_USER, [0,0], this.board),
367         x, y, x0, y0, top, depth,
368         MAX_DEPTH,
369         MAX_XDIST,
370         MAX_YDIST,
371         dyadicStack = [],
372         depthStack = [],
373         pointStack = [],
374         divisors = [], 
375         //xd_ = NaN, yd_ = NaN,
376         distOK = false,
377         j = 0;
378 
379     
380     if (this.board.updateQuality==this.board.BOARD_QUALITY_LOW) {
381         MAX_DEPTH = 12;
382         MAX_XDIST = 12;
383         MAX_YDIST = 12;
384     } else {
385         MAX_DEPTH = 20;
386         MAX_XDIST = 2;
387         MAX_YDIST = 2;
388     }
389     
390     divisors[0] = ma-mi;
391     for (i=1;i<MAX_DEPTH;i++) {
392         divisors[i] = divisors[i-1]*0.5;
393     }
394     
395     i = 1;
396     dyadicStack[0] = 1;
397     depthStack[0] = 0;
398     t = mi;
399     po.setCoordinates(JXG.COORDS_BY_USER, [this.X(t,suspendUpdate),this.Y(t,suspendUpdate)], false);
400     suspendUpdate = true;
401     x0 = po.scrCoords[1];
402     y0 = po.scrCoords[2];
403     t0 = t;
404     
405     t = ma;
406     po.setCoordinates(JXG.COORDS_BY_USER, [this.X(t,suspendUpdate),this.Y(t,suspendUpdate)], false);
407     x = po.scrCoords[1];
408     y = po.scrCoords[2];
409     
410     pointStack[0] = [x,y];
411     
412     top = 1;
413     depth = 0;
414 
415     this.points = [];
416     this.points[j++] = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x0, y0], this.board);
417     
418     do {
419         distOK = this.isDistOK(x0,y0,x,y,MAX_XDIST,MAX_YDIST)||this.isSegmentOutside(x0,y0,x,y);
420         while ( depth<MAX_DEPTH &&
421                (!distOK || depth<3 /*|| (j>1 &&!this.bendOK(xd_,yd_,x-x0,y-y0))*/) &&
422                !(!this.isSegmentDefined(x0,y0,x,y) && depth>8)
423             ) {
424             dyadicStack[top] = i;
425             depthStack[top] = depth;
426             pointStack[top] = [x,y];
427             top++;
428             
429             i = 2*i-1;
430             depth++;
431             t = mi+i*divisors[depth];
432             po.setCoordinates(JXG.COORDS_BY_USER, [this.X(t,suspendUpdate),this.Y(t,suspendUpdate)], false);
433             x = po.scrCoords[1];
434             y = po.scrCoords[2];
435             distOK = this.isDistOK(x0,y0,x,y,MAX_XDIST,MAX_YDIST)||this.isSegmentOutside(x0,y0,x,y);
436         }
437         /*
438         if (this.board.updateQuality==this.board.BOARD_QUALITY_HIGH && !this.isContinuous(t0,t,MAX_DEPTH)) {
439             //$('debug').innerHTML += 'x ';
440             this.points[j] = new JXG.Coords(JXG.COORDS_BY_SCREEN, [NaN, NaN], this.board);
441             //this.points[j] = new JXG.Coords(JXG.COORDS_BY_SCREEN, [1, 1], this.board);
442             j++;
443         }
444         */
445         this.points[j] = new JXG.Coords(JXG.COORDS_BY_SCREEN, [x, y], this.board);
446         this.updateTransform(this.points[j]);
447         j++;
448         //xd_ = x-x0;
449         //yd_ = x-y0;
450         x0 = x;
451         y0 = y;
452         t0 = t;
453         
454         top--;
455         x = pointStack[top][0];
456         y = pointStack[top][1];
457         depth = depthStack[top]+1;
458         i = dyadicStack[top]*2;
459         
460     } while (top != 0);
461     this.numberPoints = this.points.length;
462     //$('debug').innerHTML = ' '+this.numberPoints;
463     return this;
464         
465 };

467 JXG.Curve.prototype.isSegmentOutside = function (x0,y0,x1,y1) {
468     if (y0<0 && y1<0) { return true; }
469     else if (y0>this.board.canvasHeight && y1>this.board.canvasHeight) { return true; }
470     else if (x0<0 && x1<0) { return true; }
471     else if (x0>this.board.canvasWidth && x1>this.board.canvasWidth) { return true; }
472     return false;
473 };

538 JXG.Curve.prototype.updateTransform = function (p) {
539     var t, c, i, 
540         len = this.transformations.length;
541     if (len==0) {
542         return p;
543     }
544     for (i=0; i<len; i++) {
545         t = this.transformations[i];
546         t.update();
547         c = JXG.Math.matVecMult(t.matrix,p.usrCoords);
548         p.setCoordinates(JXG.COORDS_BY_USER,[c[1],c[2]]);
549     }
550     return p;
551 };

553 JXG.Curve.prototype.addTransform = function (transform) {
554     var list, i, len;
555     if (JXG.isArray(transform)) {
556         list = transform;
557     } else {
558         list = [transform];
559     }
560     len = list.length;
561     for (i=0; i<len; i++) {
562         this.transformations.push(list[i]);
563     }
564     return this;
565 };

582 /**
583  * Converts the GEONExT syntax of the defining function term into JavaScript.
584  * New methods X() and Y() for the Curve object are generated, further
585  * new methods for minX() and maxX().
586  *
587  * Also, all objects whose name appears in the term are searched and
588  * the curve is added as child to these objects. (Commented out!!!!)
589  * @see Algebra
590  * @see #geonext2JS.
591  */
592 JXG.Curve.prototype.generateTerm = function (varname, xterm, yterm, mi, ma) {
593     var fx, fy;
594 
595     // Generate the methods X() and Y()
596     if (JXG.isArray(xterm)) {
597         this.dataX = xterm;
598         this.X = function(i) { return this.dataX[i]; };
599         this.curveType = 'plot';
600         this.numberPoints = this.dataX.length;
601     } else {
602         this.X = JXG.createFunction(xterm,this.board,varname);
603         if (JXG.isString(xterm)) { 
604             this.curveType = 'functiongraph'; 
605         } else if (JXG.isFunction(xterm) || JXG.isNumber(xterm)) {
606             this.curveType = 'parameter';
607         }
608     }
609 
610     if (JXG.isArray(yterm)) {
611         this.dataY = yterm;
