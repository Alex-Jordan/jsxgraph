 42         /**
 43          * Solves a system of linear equations given by A and b using the Gauss-Jordan-elimination.
 44          * The algorithm runs in-place. I.e. the entries of A and b are changed.
 45          * @param {Array} A Square matrix represented by an array of rows, containing the coefficients of the lineare equation system.
 46          * @param {Array} b A vector containing the linear equation system's right hand side.
 47          * @throws {Error} If a non-square-matrix is given or if b has not the right length or A's rank is not full.
 48          * @returns {Array} A vector that solves the linear equation system.
 49          */
 50         Gauss: function(A, b) {
 51             var eps = JXG.Math.eps,
 52                 // number of columns of A
 53                 n = A.length>0 ? A[0].length : 0,
 54                 // copy the matrix to prevent changes in the original
 55                 Acopy,
 56                 // solution vector, to prevent changing b
 57                 x,
 58                 i, j, k,
 59                 // little helper to swap array elements
 60                 swap = function(i, j) {
 61                     var temp = this[i];
 62 
 63                     this[i] = this[j];
 64                     this[j] = temp;
 65                 };
 66 
 67             if((n !== b.length) || (n !== A.length))
 68                 throw new Error("JXG.Math.Numerics.Gauss: Dimensions don't match. A must be a square matrix and b must be of the same length as A.");
 69 
 70             x = new Array(n);
 71 
 72             Acopy = A;
 73             A = JXG.Math.matrix(n, n);
 74 
 75             // initialize solution vector
 76             for (i=0; i<n; i++) {
 77                 x[i] = b[i];
 78 
 79                 // copy A
 80                 for(j=0; j<n; j++) {
 81                     A[i][j] = Acopy[i][j];
 82                 }
 83             }
 84 
 85             // Gauss-Jordan-elimination
 86             for (j=0; j < n; j++)
 87             {
 88                 for (i = n-1; i > j; i--) {
 89                     // Is the element which is to eliminate greater than zero?
 90                     if (Math.abs(A[i][j]) > eps) {
 91                         // Equals pivot element zero?
 92                         if (Math.abs(A[j][j]) < eps) {
 93                             // At least numerically, so we have to exchange the rows
 94                             swap.apply(A, [i, j]);
 95                             swap.apply(x, [i, j]);
 96                         } else {
 97                             // Saves the L matrix of the LR-decomposition. unnecessary.
 98                             A[i][j] /= A[j][j];
 99                             // Transform right-hand-side b
100                             x[i] -= A[i][j] * b[j];
101                             // subtract the multiple of A[i][j] / A[j][j] of the j-th row from the i-th.
102                             for (k = j + 1; k < n; k ++) {
103                                 A[i][k] -= A[i][j] * A[j][k];
104                             }
105                         }
106                     }
107                 }
108                 if (Math.abs(A[j][j]) < eps) { // The absolute values of all coefficients below the j-th row in the j-th column are smaller than JXG.Math.eps.
109                     throw new Error("JXG.Math.Numerics.Gauss(): The given matrix seems to be singular.");
110                 }
111             }
112 
113             this.backwardSolve(A, x, true); // return Array
114 
115             A = Acopy;
116 
117             return x;
118         },

120         /**
121          * Solves a system of linear equations given by the right triangular matrix R and vector b.
122          * @param {Array} R Right triangular matrix represented by an array of rows. All entries a_(i,j) with i < j are ignored.
123          * @param {Array} b Right hand side of the linear equation system.
124          * @param {Boolean} [canModify=false] If true, the right hand side vector is allowed to be changed by this method.
125          * @returns {Array} An array representing a vector that solves the system of linear equations.
126          */
127         backwardSolve: function(R, b, canModify) {
128             var x, m, n, i, j;
129 
130             if(canModify) {
131                 x = b;
132             } else {
133                 x = new Array(b.length);
134                 for(i = 0; i<b.length; i++) {
135                     x[i] = b[i];
136                 }
137             }
138 
139             // m: number of rows of R
140             // n: number of columns of R
141             m = R.length;
142             n = R.length>0 ? R[0].length : 0;
143 
144             for (i = m-1; i >= 0; i--) {
145                 for (j = n-1; j > i; j--) {
146                     x[i] -= R[i][j] * x[j];
147                 }
148                 x[i] /= R[i][i];
149             }
150 
151             return x;
152         },

154         /**
155          * Compute the Eigenvalues and Eigenvectors of a symmetric 3x3 matrix with the Jacobi method
156          * Adaption of a FORTRAN program by Ed Wilson, Dec. 25, 1990
157          * @param {Array} Ain A symmetric 3x3 matrix.
158          * @returns {Array} [A,V] the matrices A and V. The diagonal of A contains the Eigenvalues, V contains the Eigenvectors.
159          */
160         Jacobi: function(Ain) {
161         var i,j,k,aa,si,co,tt,eps=JXG.Math.eps,
162             sum = 0.0,
163             ssum, amax,
164             n = Ain.length,
165             V = [[0,0,0],[0,0,0],[0,0,0]],
166             A = [[0,0,0],[0,0,0],[0,0,0]];
167 
168         // Initialization. Set initial Eigenvectors.
169         for (i=0;i<n;i++) {
170             for (j=0;j<n;j++) {
171                 V[i][j] = 0.0;
172                 A[i][j] = Ain[i][j];
173                 sum += Math.abs(A[i][j]);
174             }
175             V[i][i] = 1.0;
176         }
177         // Trivial problems
178         if (n==1) { return [A,V]; }
179         if (sum<=0.0) { return [A,V]; }
180 
181         sum /= (n*n);
182 
183         // Reduce matrix to diagonal
184         do {
185             ssum = 0.0;
186             amax = 0.0;
187             for (j=1;j<n;j++) {
188                 for (i=0;i<j;i++) {
189                     // Check if A[i][j] is to be reduced
190                     aa = Math.abs(A[i][j]);
191                     if (aa > amax) {
192                         amax = aa;
193                     }
194                     ssum += aa;
195                     if (aa >= eps) {
196                         // calculate rotation angle
197                         aa = Math.atan2(2.0 * A[i][j], A[i][i] - A[j][j]) * 0.5;
198                         si = Math.sin(aa);
199                         co = Math.cos(aa);
200                         // Modify 'i' and 'j' columns
201                         for (k = 0; k < n; k++) {
202                             tt = A[k][i];
203                             A[k][i] = co * tt + si * A[k][j];
204                             A[k][j] = -si * tt + co * A[k][j];
205                             tt = V[k][i];
206                             V[k][i] = co * tt + si * V[k][j];
207                             V[k][j] = -si * tt + co * V[k][j];
208                         }
209                         // Modify diagonal terms
210                         A[i][i] = co * A[i][i] + si * A[j][i];
211                         A[j][j] = -si * A[i][j] + co * A[j][j];
212                         A[i][j] = 0.0;
213                         // Make 'A' matrix symmetrical
214                         for (k = 0; k < n; k++) {
215                             A[i][k] = A[k][i];
216                             A[j][k] = A[k][j];
217                         }
218                         // A[i][j] made zero by rotation
219                     }
220                 }
221             }
222         } while(Math.abs(ssum)/sum>eps);
223         return [A,V];
224     },

226         /**
227          * Calculates the integral of function f over interval using Newton-Cotes-algorithm.
228          * @param {Array} interval The integration interval, e.g. [0, 3].
229          * @param {function} f A function which takes one argument of type number and returns a number.
230          * @param {object} [config={number_of_nodes:28,integration_type:'milne'}] The algorithm setup. Accepted properties are number_of_nodes of type number and integration_type
231          * with value being either 'trapez', 'simpson', or 'milne'.
232          * @returns {Number} Integral value of f over interval
233          * @throws {Error} If config.number_of_nodes doesn't match config.integration_type an exception is thrown. If you want to use
234          * simpson rule respectively milne rule config.number_of_nodes must be dividable by 2 respectively 4.
235          * @example
236          * function f(x) {
237          *   return x*x;
238          * }
239          *
240          * // calculates integral of <tt>f</tt> from 0 to 2.
241          * var area1 = JXG.Math.Numerics.NewtonCotes([0, 2], f);
242          *
243          * // the same with an anonymous function
244          * var area2 = JXG.Math.Numerics.NewtonCotes([0, 2], function (x) { return x*x; });
245          *
246          * // use trapez rule with 16 nodes
247          * var area3 = JXG.Math.Numerics.NewtonCotes([0, 2], f,
248          *                                   {number_of_nodes: 16, intergration_type: 'trapez'});
249          */
250         NewtonCotes: function(interval, f, config) {
251             var integral_value = 0.0,
252                 number_of_nodes = config && typeof config.number_of_nodes === 'number' ? config.number_of_nodes : 28,
253                 available_types = {trapez: true, simpson: true, milne: true},
254                 integration_type = config && config.integration_type && available_types.hasOwnProperty(config.integration_type) && available_types[config.integration_type] ? config.integration_type : 'milne',
255                 step_size = (interval[1] - interval[0]) / number_of_nodes,
256                 evaluation_point, i, number_of_intervals;
257 
258             switch(integration_type) {
259                 case 'trapez':
260                     integral_value = (f(interval[0]) + f(interval[1])) * 0.5;
261 
262                     evaluation_point = interval[0];
263                     for (i = 0; i < number_of_nodes - 1; i++)
264                     {
265                         evaluation_point += step_size;
266                         integral_value   += f(evaluation_point);
267                     }
268                     integral_value *= step_size;
269 
270                     break;
271                 case 'simpson':
272                     if (number_of_nodes%2 > 0) {
273                         throw new Error("JSXGraph:  INT_SIMPSON requires config.number_of_nodes dividable by 2.");
274                     }
275                     number_of_intervals = number_of_nodes / 2.0;
276                     integral_value = f(interval[0]) + f(interval[1]);
277                     evaluation_point = interval[0];
278                     for (i = 0; i < number_of_intervals - 1; i++)
279                     {
280                         evaluation_point += 2.0 * step_size;
281                         integral_value   += 2.0 * f(evaluation_point);
282                     }
283                     evaluation_point = interval[0] - step_size;
284                     for (i = 0; i < number_of_intervals; i++)
285                     {
286                         evaluation_point += 2.0 * step_size;
287                         integral_value   += 4.0 * f(evaluation_point);
288                     }
289                     integral_value *= step_size / 3.0;
290                     break;
291                 default:
292                     if (number_of_nodes%4 > 0) {
293                         throw new Error("JSXGraph: Error in INT_MILNE: config.number_of_nodes must be a multiple of 4");
294                     }
295                     number_of_intervals = number_of_nodes * 0.25;
296                     integral_value = 7.0 * (f(interval[0]) + f(interval[1]));
297                     evaluation_point = interval[0];
298                     for (i = 0; i < number_of_intervals - 1; i++)
299                     {
300                         evaluation_point += 4.0 * step_size;
301                         integral_value   += 14.0 * f(evaluation_point);
302                     }
303                     evaluation_point = interval[0] - 3.0 * step_size;
304                     for (i = 0; i < number_of_intervals; i++)
305                     {
306                         evaluation_point += 4.0 * step_size;
307                         integral_value   += 32.0 * (f(evaluation_point) + f(evaluation_point + 2 * step_size));
308                     }
309                     evaluation_point = interval[0] - 2.0 * step_size;
310                     for (i = 0; i < number_of_intervals; i++)
311                     {
312                         evaluation_point += 4.0 * step_size;
313                         integral_value   += 12.0 * f(evaluation_point);
314                     }
315                     integral_value *= 2.0 * step_size / 45.0;
316             }
317             return integral_value;
318         },

331         /**
332          * Newton's method to find roots of a funtion in one variable.
333          * @param {function} f We search for a solution of f(x)=0.
334          * @param {Number} x initial guess for the root, i.e. staring value.
335          * @param {object} object optional object that is treated as "this" in the function body. This is useful, if the function is a
336          *                 method of an object and contains a reference to its parent object via "this".
337          * @returns {Number} A root of the function f.
338          */
339         Newton: function(f, x, object) {
340             var i = 0,
341                 h = 0.000001,
342                 newf = f.apply(object,[x]), // set "this" to "object" in f
343                 df;
344             while (i<50 && Math.abs(newf)>h) {
345                 df = this.D(f,object)(x);
346                 if (Math.abs(df)>h) {
347                     x -= newf/df;
348                 } else {
349                     x += (Math.random()*0.2-1.0);
350                 }
351                 newf = f.apply(object,[x]);
352                 i++;
353             }
354             return x;
355         },

373 /**
374  * Calculates second derivatives at the knots.
375  * @param {Array} x x values of knots
376  * @param {Array} y y values of knots
377  * @returns {Array} Second derivatives of the interpolated function at the knots.
378  * @see #splineEval
379  */
380 JXG.Math.Numerics.splineDef = function(x, y) {
381     var n = x.length,
382         pair, i, diag, z, l,
383         data = new Array(),
384         dx = [],
385         delta = [],
386         F;
387 
388     if (n != y.length)
389         throw new Error("JSXGraph: Error in JXG.Math.Numerics.splineDef: Input vector dimensions do not match.");
390 
391     for (i=0; i<n; i++) {
392         pair = {X: x[i], Y: y[i]};
393         data.push(pair);
394     }
395     data.sort(function (a,b) { return a.X - b.X; });
396     for (i=0; i<n; i++) {
397         x[i] = data[i].X;
398         y[i] = data[i].Y;
399     }
400 
401     for (i=0; i<n-1; i++) {
402         dx.push(x[i+1] - x[i]);
403     }
404     for (i=0; i<n-2; i++) {
405         delta.push(6 * (y[i+2] - y[i+1])/(dx[i+1]) - 6 * (y[i+1] - y[i])/(dx[i]));
406     }
407 
408     // ForwardSolve
409     diag = new Array();
410     z = new Array();
411     diag.push(2*(dx[0] + dx[1]));
412     z.push(delta[0]);
413 
414     for (i=0; i<n-3; i++) {
415         l = dx[i+1]/diag[i];
416         diag.push(2 * (dx[i+1] + dx[i+2]) - l*dx[i+1]);
417         z.push(delta[i+1] - l*z[i]);
418     }
419 
420     // BackwardSolve
421     F = new Array();
422     F[n-3] = z[n-3]/diag[n-3];
423     for(i=n-4; i>=0; i--) {
424         F[i] = (z[i] - (dx[i+1]*F[i+1]))/diag[i];
425     }
426 
427     // Generate f''-Vector
428     for(i=n-3; i>=0; i--)
429         F[i+1] = F[i];
430 
431     // natural cubic spline
432     F[0] = 0;
433     F[n-1] = 0;
434     return F;
435 };

437 /**
438  * Evaluate points on spline.
439  * @param {Number,Array} x0 A single float value or an array of values to evaluate
440  * @param {Array} x x values of knots
441  * @param {Array} y y values of knots
442  * @param {Array} F Second derivatives at knots, calculated by {@link #splineDef}
443  * @see #splineDef
444  * @returns {Number,Array} A single value or an array, depending on what is given as x0.
445  */
446 JXG.Math.Numerics.splineEval = function(x0, x, y, F) {
447     var n = x.length,
448         l = 1,
449         asArray = false,
450         y0, i, j, a, b, c, d, x_;
451 
452     if (n != y.length)
453         throw new Error("JSXGraph: Error in JXG.Math.Numerics.splineEval: Defining vector dimensions do not match.");
454     
455     // number of points to be evaluated
456     if(JXG.isArray(x0)) {
457         l = x0.length;
458         asArray = true;
459     } else
460         x0 = [x0];
461     
462     y0 = new Array();
463     
464     for (i=0; i<l; i++) {
465         // is x0 in defining interval?
466         if( (x0[i] < x[0]) || (x[i] > x[n-1]))
467             return 'NaN';
468         
469         // determine part of spline in which x0 lies
470         for (j=1; j<n; j++) {
471             if (x0[i] <= x[j])
472                 break;
473         }
474         j--;
475         
476         // we're now in the j-th partial interval, i.e. x[j] < x0[i] <= x[j+1];
477         // determine the coefficients of the polynomial in this interval
478         a = y[j];
479         b = (y[j+1]-y[j])/(x[j+1]-x[j]) - (x[j+1]-x[j])/6 * (F[j+1]+2*F[j]);
480         c = F[j]/2;
481         d = (F[j+1]-F[j])/(6*(x[j+1]-x[j]));
482         // evaluate x0[i]
483         x_ = x0[i]-x[j];
484         //y0.push(a + b*x_ + c*x_*x_ + d*x_*x_*x_);
485         y0.push(a + (b + (c+ d*x_)*x_)*x_);
486     }
487 
488     if (asArray)
489         return y0;
490     else
491         return y0[0];
492 };

800 /**
801  * Computes the B-spline curve of order k (order = degree+1) through a given set of points.
802  * @param data array consisting of JXG.Points.
803  * @param integer order of the B-spline curve.
804  * @todo: closed B-spline curves
805  * @type {function}
806  * @return {function} A function of one parameter t which returns the coordinates of the B-spline curve in t.
807  */
808 JXG.Math.Numerics.bspline = function(points, order) {
809     var knots = [], N = [];
810     
811     var _knotVector = function(n,k){
812             var j, kn = [];
813             for (j=0;j<n+k+1;j++) {
814                 if (j<k) {
815                     kn[j] = 0.0;
816                 } else if (j<=n) {
817                     kn[j] = j-k+1;
818                 } else {
819                     kn[j] = n-k+2;
820                 }
821             }
822             return kn;
823         };
824 
825     var _evalBasisFuncs = function(t,kn,n,k,s) {
826             var i,j,s,a,b,den,
827                 N = [];
828                 
829             if (kn[s]<=t && t<kn[s+1]) {
830                 N[s]=1; 
831             } else {
832                 N[s]=0; 
833             }
834             for (i=2;i<=k;i++) {
835                 for (j=s-i+1;j<=s;j++) {
836                     if (j<=s-i+1||j<0) {
837                         a = 0.0;
838                     } else {
839                         a = N[j];
840                     }
841                     if (j>=s) {
842                         b = 0.0;
843                     } else {
844                         b = N[j+1];
845                     }
846                     den = kn[j+i-1]-kn[j];
847                     if (den==0) {
848                         N[j] = 0;
849                     } else {
850                         N[j] = (t-kn[j])/den*a;
851                     }
852                     den = kn[j+i]-kn[j+1];
853                     if (den!=0) {
854                         N[j] += (kn[j+i]-t)/den*b;
855                     }
856                 }
857             }    
858             return N;
859         };
860 
861 
862     return [function(t,suspendedUpdate) {
863                 var len = points.length, 
864                     y, i, j, s, a, b,
865                     n = len-1,
866                     k = order;
867                     
868                 if (n<=0) return NaN;
869                 if (n+2<=k) k = n+1;
870                 if (t<=0) return points[0].X(); 
871                 if (t>=n-k+2) return points[n].X();
872                 
873                 knots = _knotVector(n,k);
874                 s = Math.floor(t)+k-1;
875                 N = _evalBasisFuncs(t,knots,n,k,s);
876                 
877                 y = 0.0;
878                 for (j=s-k+1;j<=s;j++) {
879                     if (j<len && j>=0) y += points[j].X()*N[j];
880                 }
881                 return y;
882             },
883             function(t,suspendedUpdate) {
884                 var len = points.length, 
885                     y, i, j, s, a, b,
886                     n = len-1,
887                     k = order;
888                     
889                 if (n<=0) return NaN;
890                 if (n+2<=k) k = n+1;
891                 if (t<=0) return points[0].Y(); 
892                 if (t>=n-k+2) return points[n].Y();
893 
894                 knots = _knotVector(n,k);
895                 s = Math.floor(t)+k-1;
896                 N = _evalBasisFuncs(t,knots,n,k,s);
897                 
898                 y = 0.0;
899                 for (j=s-k+1;j<=s;j++) {
900                     if (j<len && j>=0) y += points[j].Y()*N[j];
901                 }
902                 return y;
903             }, 
904             0, function() {return points.length-1;}
905            ];

940     var xarr,yarr,i,delta,j,x,y,x1,delta1,y1;
941     
942     xarr = [];

945     x = start;

952

954             if (type=='right') {
955                 y = f(x+delta);

958             } else if ((type=='left') || (type=='trapezodial')) {
959                 y = f(x);
960             } else if (type=='lower') {
961                 y = f(x);
962                 for (x1=x+delta1;x1<=x+delta;x1+=delta1) {
963                     y1 = f(x1);
964                     if (y1<y) { y = y1; }
965                 }
966             } else { // (type=='upper')
967                 y = f(x);

972             }
973

978                 y = f(x);
979             }
980             xarr[j] = x; yarr[j] = y;
981             j++;
982             xarr[j] = x; yarr[j] = 0.0;
